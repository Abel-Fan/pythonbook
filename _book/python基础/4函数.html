4 函数

4.1 函数的语法

def 函数名([参数]):
    """函数的简介.
    【ps:这行加空格】
    函数的详细信息.
    """
    函数体
    [return]
注释
函数名.__doc__
参数分类
1、必选参数
2、默认参数
3、可变参数
4、关键字参数
必选参数
#必须要传参
def fn(a,b):
    print("a:",a)
    print("b:",b)
fn(10,20)     # a=10,b=20 
fn(b=10,a=20) # b=10,a=20 
fn()   # 错误
fn(10) # 错误
默认参数
# 缺省参数
def fn(a,b,c=10):
    print("a:",a)
    print("b:",b)
    print("c:",c)
fn(10,20)  # a=10 b=20 c=10
fn(10,20,30) # a=10 b=20 c=30
fn(c=30,b=20,a=10) # a=10 b=20 c=30
fn(c=30,20,10) #错误
# 参数可以是变量，但是只能挂载一次
num = 10
def fn(a, b, c=num):
    print("a:", a) # 10 
    print("b:", b) # 20
    print("c:", c) # 10
num = 20
fn(10,20)
#可变参数实现数据共享
mylist = []
def fn(a,arr=[]):
    arr.append(a)
    print(arr)
fn(10) #[10]
fn(20) #[10,20]
fn(30) #[10,20,30]
可变参数
# 接受任意个数的参数，内部是元组形式
def fn(a,b,c=44,*args):
print("a:",a)  # 10
print("b:",b)  # 20
print("c:",c)  # 30
print(args)    # (40,)
fn(10,20,30,40)
关键字参数
# 以键值的形式传参，内容是字典形式
可变参数，关键字参数的数据解压
# 列表和字典数据型式 -> 数据压缩
# 列表解压
def fn(*args):
    print(args)  
list1 = [1,2,3,4,5]
fn(*list1)  #(1,2,3,4,5)
# 字典解压
xh = {'name':'xh','sex':'女'}
def fn(**kwargs):
    print(kwargs)
list1 = [1,2,3,4,5]
fn(**xh)
传参的顺序
必选参数、默认参数、可变参数、关键字参数
返回值
return作用：返回并且终止函数
return注意：
    1、可以写多个return语句，最终只有一个被执行
    2、不能再return语句之后书写代码
    3、return只能返回一个数据，如果想返回多个数据必须为list\tuple\dict

匿名函数

作为参数进行传递，简化程序
语法
lambda a[,b] : 函数体
参数可以有多个，之间","分割 
函数体只有一个表达式，并且自动return
调用方式
1、作为参数直接传递(常用)
2、用变量进行保存（fn = lambda a,b,c: a+b+c）
3、自调用  (lambda a,b,c: a+b+c)(1,2,3)
举例
fn = lambda a,b,c: a+b+c
print(fn(1,2,3))
# 类似于
def fn(a,b,c):
    return a+b+c

高阶函数

把一个函数名作为参数传递到另外一个函数中(实参高阶函数)
把一个函数名作为返回值返回(返回值高阶函数)
常用的高阶函数
map(fn,iter)
map(fn,iter1,iter2)
filter(fn,iter)
reduce(fn,iter)
map
list1 = [1,2,3,4,5]
list2 = [1,2,3,4,5]
print(list(map(lambda x,y:x+y,list1,list2)))
# [2,4,6,8,10]
print(list(map(lambda x:x**2,list1)))
# [1,4,9,16,25]
filter
list1 = [1,2,3,4,5]
print(list(filter(lambda x:x%2==0,list1)))
reduce
from functools import reduce
list1 = [1,2,3,4,5]
def fn(x,y):
    print("x:",x,end="  ")
    print("y:",y,end="  ")
    print("x+y",x+y)
    return x+y
reduce(fn,list1)

局部环境全局环境

函数内称为局部环境，局部环境内的变量就是局部变量
整个.py文件中为全局环境，全局变量在整个.py中都能访问变量
通过dir()方法可以查看全局环境中的变量以及局部环境中的变量
globel nonlocal 语句
globel 全局变量声明（不能申明的同时赋值）
nonlocal 非局部非全局声明

闭包函数

当两个函数发生嵌套关系时，内层函数调用了外层函数的变量。当在外部调用内层函数时，此刻发生了闭包。

递归

自己调用自己
本质 相当于循环
必须有终止

装饰器

定义
给一个函数增加新的功能
应用场景
当一个应用上线，如果想要增加他的功能，就希望不要过多的改变源代码。
条件
1、不能改变源代码
2、不能改变源代码的调用方式
3、满足1和2基础上来增加新功能
公式
<函数+闭包+高阶函数+语法糖=装饰器>
柯里化
def fn(a):
    def fn1(b):
        def fn2(c):
            print(a+b+c)
        return fn2
    return fn1
fn(10)(20)(30)
装饰器传参
import time
def timer(a,b):
    def timer1(fn):
        def newfn():
            start = time.time()
            fn()
            end = time.time()
            print(a,b)
            print("运行总时间:%s"%(end-start))
        return newfn
    return timer1
def starttime(fn):
    def newfn():
        print("开始时间：%s"%time.time())
        fn()
    return newfn 
@starttime
@timer(10,20)
def fn():
    time.sleep(1)
    print("程序运行")
fn()
